Я ціную активний спосіб життя та завжди стимулюю себе до досягнення нових спортивних досягнень.

Програмування

function findMinDivisor(num, div) {
    // якщо дільник не заданий, встановлюємо його як 2
    div = div || 2;

    // якщо число менше 2, повертаємо null
    if (num < 2) {
        return null;
    }

    // якщо дільник більший або дорівнює числу, повертаємо саме число як мінімальний дільник
    if (div >= num) {
        return num;
    }

    // якщо число ділиться на дільник без залишку, повертаємо його як мінімальний дільник
    if (num % div === 0) {
        return div;
    }

    // інакше збільшуємо дільник на 1 та рекурсивно викликаємо функцію
    return findMinDivisor(num, div + 1);
}
